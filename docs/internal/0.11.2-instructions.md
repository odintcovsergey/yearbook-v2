```

## Логика overflow (из build_album.jsx, строки 598-695)

### Лайт (basePages=4)
- **25-27** (overflow 1-3): 4 базовых + 1× L-Overflow-Row (`students≤3` + classphotoframe)
- **28-30** (overflow 4-6): 4 базовых + 1× extra обычная сетка (5-я страница, неполная)
- **31-32** (overflow 7-8): 4 базовых + 1× полная extra сетка + 1× L-Overflow-Row-Right (`students≤3` + classphotoframe)

### Мини (basePages=2)
- **25-28** (overflow 1-4): 2 базовых + 1× N-Overflow-Row (`students≤4` + classphotoframe)
- **29-36** (overflow 5-12): 2 базовых + 1× extra обычная сетка (3-я страница, неполная)

### Универсальный алгоритм
```
overflow = total - basePages * slotsPerPage
rowCapacity = master.slot_capacity.students  (3 для L, 4 для N)

if overflow <= 0:
  return  // всё в базовых
elif overflow <= rowCapacity:
  ветка single_row (один overflow-row мастер)
elif overflow <= slotsPerPage:
  ветка extra_grid (одна обычная сетка с пустыми слотами)
else:
  ветка grid_plus_row (полная сетка + overflow-row-right; только Лайт 31-32)
```

### Чётность страниц для extra_grid
- Лайт: страница 5 → нечётная (1-based) → leftMaster
- Мини: страница 3 → нечётная → leftMaster

То есть extra_grid всегда использует `leftMaster`.

## Состояние БД (проверено в 0.11.1 разведке)

- L-Overflow-Row: `slot_capacity={students:3, photos_full:1}`, page_role='student_overflow', applies_to=['standard','universal','maximum','medium','light','mini','individual']
- L-Overflow-Row-Right: то же
- N-Overflow-Row: `slot_capacity={students:4, photos_full:1}`, page_role='student_overflow', applies_to=общеупотребимый

Никаких миграций БД не требуется.

/**
 * Конфигурация overflow для adaptive_grid комплектаций (Лайт/Мини).
 *
 * Применяется когда total > base_pages * slotsPerPage.
 *
 * Алгоритм веток (в buildAdaptiveGridStudents):

export type AdaptiveGridOverflow = {
  /** Фильтр для overflow-row LEFT мастера (L-Overflow-Row, N-Overflow-Row). */

   * Опционально, нужно только для Лайт 31-32. Если не задано и overflow
   * требует grid+row — пишется warning master_not_found.
   */
  row_right_filter?: MasterFilter;
};
```

### 2.2 Заменить overflow-блок в `buildAdaptiveGridStudents`

**Текущая логика в 0.11.1** (degraded — обрезка):
```typescript
const maxOnBase = basePages * slotsPerPage;
if (total > maxOnBase) {
  pushWarning(students_overflow, "..."); 
}
const studentsToPlace = total > maxOnBase ? students.slice(0, maxOnBase) : students;
// ... базовые страницы
```

**Стало:**
```typescript
const maxOnBase = basePages * slotsPerPage;
const overflowCount = total - maxOnBase;

// Базовые страницы — заполняем как и было
const baseStudents = total > maxOnBase ? ctx.input.students.slice(0, maxOnBase) : ctx.input.students;
const hasQuote = section.has_quote === true;

for (let i = 0; i < basePages; i++) {
  const slice = baseStudents.slice(i * slotsPerPage, (i + 1) * slotsPerPage);
  if (slice.length === 0) break;
  const isLeft = i % 2 === 0;
  const master = isLeft ? leftMaster : rightMaster;

  ctx.spreads.push({
    spread_index: ctx.spreadCounter.value++,
    template_id: master.id,
    template_name: master.name,
    data: buildGridStudentData(slice, slotsPerPage, hasQuote),
  });
}

// Overflow-логика
if (overflowCount <= 0) return;

const ovf = section.overflow;
if (!ovf) {
  pushWarning(ctx, {
    code: 'students_overflow',
    detail: `${ctx.config.config_type}: total=${total} > maxOnBase=${maxOnBase}, overflow_strategy не задана`,
  });
  return;
}

// Найти overflow-row мастер чтобы узнать rowCapacity
const rowFilter = { ...ovf.row_filter, applies_to_config: ctx.config.config_type };
const rowR = findMaster(ctx.config.template_set.spreads, rowFilter);
if (!rowR.ok) {
  pushWarning(ctx, {
    code: 'master_not_found',
    detail: `overflow row: ${ovf.row_filter.expected_name_hint ?? '?'}`,
  });
  return;
}
if (rowR.warning) pushWarning(ctx, rowR.warning);
const rowCapacity = rowR.master.slot_capacity?.students ?? 0;

const overflowStudents = ctx.input.students.slice(maxOnBase);

if (overflowCount <= rowCapacity) {
  // Ветка single_row
  buildOverflowRow(ctx, rowR.master, overflowStudents, rowCapacity);
} else if (overflowCount <= slotsPerPage) {
  // Ветка extra_grid: одна неполная сетка на нечётной странице (всегда leftMaster)
  ctx.spreads.push({
    spread_index: ctx.spreadCounter.value++,
    template_id: leftMaster.id,
    template_name: leftMaster.name,
    data: buildGridStudentData(overflowStudents, slotsPerPage, hasQuote),
  });
} else {
  // Ветка grid_plus_row (только Лайт 31-32)
  if (!ovf.row_right_filter) {
    pushWarning(ctx, {
      code: 'master_not_found',
      detail: `overflow grid_plus_row: row_right_filter не задан в scenario, overflowCount=${overflowCount}`,
    });
    return;
  }
  const rightF = { ...ovf.row_right_filter, applies_to_config: ctx.config.config_type };
  const rightR = findMaster(ctx.config.template_set.spreads, rightF);
  if (!rightR.ok) {
    pushWarning(ctx, {
      code: 'master_not_found',
      detail: `overflow row_right: ${ovf.row_right_filter.expected_name_hint ?? '?'}`,
    });
    return;
  }
  if (rightR.warning) pushWarning(ctx, rightR.warning);

  // Полная extra-сетка
  const fullSlice = overflowStudents.slice(0, slotsPerPage);
  ctx.spreads.push({
    spread_index: ctx.spreadCounter.value++,
    template_id: leftMaster.id,
    template_name: leftMaster.name,
    data: buildGridStudentData(fullSlice, slotsPerPage, hasQuote),
  });

  // Overflow-row-right
  const remainSlice = overflowStudents.slice(slotsPerPage);
  const rightCapacity = rightR.master.slot_capacity?.students ?? rowCapacity;
  buildOverflowRow(ctx, rightR.master, remainSlice, rightCapacity);
}
```

### 2.3 Новая функция `buildOverflowRow`

Добавить после `buildAdaptiveGridStudents`:

```typescript
/**
 * Сборка overflow-row страницы для Лайт/Мини.
 *
 * Заполняет:
 *  - studentportrait_N, studentname_N для всех students (без quote — Лайт/Мини без цитат)
 *  - null'ы для пустых слотов
 *  - classphotoframe из common_photos.full_class[0] (если есть placeholder)
 */
function buildOverflowRow(
  ctx: BuildContext,
  master: SpreadTemplate,
  students: Student[],
  rowCapacity: number,
): void {
  const data = buildGridStudentData(students, rowCapacity, /* hasQuote = */ false);

  if (hasPlaceholder(master, 'classphotoframe')) {
    const fc = ctx.input.common_photos.full_class;
    if (fc.length >= 1) {
      data.classphotoframe = fc[0];
    } else {
      data.classphotoframe = null;
      pushWarning(ctx, {
        code: 'class_photo_missing',
        detail: `master ${master.name} has classphotoframe placeholder but common_photos.full_class is empty`,
      });
    }
  }

  ctx.spreads.push({
    spread_index: ctx.spreadCounter.value++,
    template_id: master.id,
    template_name: master.name,
    data,
  });
}
```
